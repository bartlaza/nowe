void wyczysc (char plansza[3][3])
{

}

int pelna(char plansza[3][3])
{
int i, j;
    for (i=0;i<3;i++)
        for (j=0;j<3;j++)
            if (plansza[i][j] == ' ')
                return 0;
    return 1;
 
}
